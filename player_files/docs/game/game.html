Bytefight 2026 Game Engine - King of the Hill

A competitive coding game where players control territories, manage stamina, and compete for control of hills on a dynamic battlefield. The files included in these packages are sufficient for simulating the entirety of the game mechanics. If you are a competitor, you will most likely only be looking at files in this package (and potentially some scripts if you are running from terminal).

If you're confused about how the game works, the apply_turn, _execute_move, and _execute_paint functions of the Board.board class is an excellent place to get started.

If you are a developer, for game setup and gameplay management, see the game_runner package.

 442    def apply_turn(self, player_parity: int, actions: Action | Iterable[Action]) -> bool:
 443        """
 444        Apply a full turn consisting of one or more actions.
 445
 446        Actions are executed sequentially until completion, death,
 447        or invalid execution.
 448
 449        Args:
 450            player_parity (int): Acting player's parity.
 451            actions (Action or Iterable[Action]): Single action or iterable of actions.
 452
 453        Returns:
 454            bool: True if at least one move was successfully executed.
 455        """
 456        # convert actions to iterable if they aren't already
 457        if isinstance(actions, (Action.Move, Action.Paint)):
 458            action_iterable = [actions]
 459        else:
 460            try:
 461                action_iterable = list(actions)
 462            except TypeError:
 463                action_iterable = [actions]
 464        
 465        # keeps track of the number of moves thave have already executed this turn
 466        moves_this_turn = 0
 467        for action in action_iterable:
 468            if isinstance(action, Action.Move):
 469                if not self._execute_move(player_parity, action, moves_this_turn):
 470                    return False
 471                moves_this_turn += 1
 472            elif isinstance(action, Action.Paint):
 473                if not self._execute_paint(player_parity, action):
 474                    return False
 475            else:
 476                continue
 477            
 478            if self.get_player(player_parity).is_dead():
 479                return False
 480            
 481            if self.get_opponent(player_parity).is_dead():
 482                return moves_this_turn > 0
 483        
 484        self.end_turn()
 485
 486        return moves_this_turn > 0

 538    def _execute_move(self, player_parity: int, move: Action.Move, moves_this_turn: int) -> bool:
 539        """
 540        Execute a movement action for a player.
 541
 542        Handles stamina costs, beacon travel, collisions, erase effects,
 543        power-ups, and beacon placement.
 544
 545        Args:
 546            player_parity (int): Acting player's parity.
 547            move (Action.Move): Move action.
 548            moves_this_turn (int): Number of moves already taken.
 549
 550        Returns:
 551            bool: True if the move was successful.
 552        """
 553        player = self.get_player(player_parity)
 554        self._apply_powerup_if_present(player_parity, self.cells[player.loc.r][player.loc.c])
 555
 556        if moves_this_turn >= 1:
 557            cost = player.stamina < GameConstants.EXTRA_MOVE_COST * moves_this_turn
 558            if player.stamina < cost:
 559                return False 
 560            player.stamina -= cost
 561
 562        if move.move_type == MoveType.ERASE:
 563            cost = GameConstants.ERASE_STEP_EXTRA_COST
 564            if player.stamina < cost:
 565                return False 
 566            player.stamina -= cost
 567            
 568        if move.move_type == MoveType.BEACON_TRAVEL:
 569            # use beacon to travel 
 570            target_loc = self._beacon_travel(player_parity, move, moves_this_turn)
 571            if target_loc is None:
 572                return False
 573        else:
 574            # don't use beacon to travel
 575            target_loc = player.loc + move.direction
 576            if self.oob(target_loc):
 577                return False
 578        
 579        # movement occurs here
 580        target_cell = self.cells[target_loc.r][target_loc.c]
 581        if target_cell.is_wall:
 582            return False
 583
 584        if self._resolve_collision(player_parity):
 585            return True
 586
 587        previous_loc = player.loc
 588        player.loc = target_loc
 589        
 590        # handle movement effects
 591        self._handle_erase_effects(player_parity, target_cell, move.move_type)
 592        self._apply_powerup_if_present(player_parity, target_cell)
 593        if move.place_beacon:
 594            self._place_beacon(player_parity, target_loc)
 595
 596        #  important that this comes before collision resolution
 597        if not player.clamp_stamina():
 598            return False
 599        
 600        
 601        
 602        return True

 676        """
