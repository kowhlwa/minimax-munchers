 124    def paint(self, player_parity: int, max_value: int) -> None:
 125        """
 126        Apply paint to the cell for a given player.
 127
 128        Paint can only be applied if there is no active beacon and the
 129        cell is either unowned or already owned by the painting player.
 130        The paint value has a maximum maginitude specified in GameConstants.
 131
 132        Args:
 133            player_parity (int): Parity of the painting player.
 134            max_value (int): Maximum absolute value for paint strength.
 135        """
 136        if (
 137            Parity.unowned(self.beacon_parity) and (Parity.unowned(self.paint_value) or Parity.owned(self.paint_value, player_parity))
 138        ):
 139            self.paint_value  = min(max(self.paint_value + player_parity, -max_value),  max_value)

 141    def weaken_opponent(self, player_parity: int) -> None:
 142        """
 143        Reduce the opponent's paint strength on this cell.
 144
 145        This can only occur if there is no active beacon and the cell
 146        is currently owned by the opposing player. Occurs automatically after moving into a cell.
 147
 148        Args:
 149            player_parity (int): Parity of the acting player.
 150        """
 151        if (
 152            Parity.unowned(self.beacon_parity) and Parity.owned(self.paint_value, Parity.get_opponent_parity(player_parity))
 153        ):
 154            self.paint_value += player_parity

 156    def erase(self) -> None:
 157        if(Parity.unowned(self.beacon_parity)):
 158            self.paint_value = 0

 316    def apply_bid(self, bid1:int, bid2:int):
 317        # assumes given bids are valid
 318        """
 319        Resolve bidding between players to determine turn order.
 320
 321        The higher bid gains initiative and pays stamina equal to the bid.
 322        Ties are resolved randomly.
 323
 324        Args:
 325            bid1 (int): Player 1's bid.
 326            bid2 (int): Player 2's bid.
 327        """
 328        if not self.is_valid_bid(bid1) or not self.is_valid_bid(bid2):
 329            return
 330        
 331        if(bid1 > bid2):
 332            self.parity_to_play = 1
 333            self.p1.stamina -= bid1
 334        elif(bid1 < bid2):
 335            self.parity_to_play = -1
 336            self.p2.stamina -= bid2
 337        else:
 338            if random.random() < 0.5:
 339                self.parity_to_play = 1
 340                self.p1.stamina -= bid1
 341            else:
 342                self.parity_to_play = -1
 343                self.p2.stamina -= bid2

 441    def apply_turn(self, player_parity: int, actions: Action | Iterable[Action]) -> bool:
 442        """
 443        Apply a full turn consisting of one or more actions.
 444
 445        Actions are executed sequentially until completion, death,
 446        or invalid execution.
 447
 448        Args:
 449            player_parity (int): Acting player's parity.
 450            actions (Action or Iterable[Action]): Single action or iterable of actions.
 451
 452        Returns:
 453            bool: True if at least one move was successfully executed.
 454        """
 455        # convert actions to iterable if they aren't already
 456        if isinstance(actions, (Action.Move, Action.Paint)):
 457            action_iterable = [actions]
 458        else:
 459            try:
 460                action_iterable = list(actions)
 461            except TypeError:
 462                action_iterable = [actions]
 463        
 464        # keeps track of the number of moves thave have already executed this turn
 465        moves_this_turn = 0
 466        for action in action_iterable:
 467            if isinstance(action, Action.Move):
 468                if not self._execute_move(player_parity, action, moves_this_turn):
 469                    return False
 470                moves_this_turn += 1
 471            elif isinstance(action, Action.Paint):
 472                if not self._execute_paint(player_parity, action):
 473                    return False
 474            else:
 475                continue
 476            
 477            if self.get_player(player_parity).is_dead():
 478                return False
 479            
 480            if self.get_opponent(player_parity).is_dead():
 481                return moves_this_turn > 0
 482        
 483        self.end_turn()
 484
 485        return moves_this_turn > 0

 537    def _execute_move(self, player_parity: int, move: Action.Move, moves_this_turn: int) -> bool:
 538        """
 539        Execute a movement action for a player.
 540
 541        Handles stamina costs, beacon travel, collisions, erase effects,
 542        power-ups, and beacon placement.
 543
 544        Args:
 545            player_parity (int): Acting player's parity.
 546            move (Action.Move): Move action.
 547            moves_this_turn (int): Number of moves already taken.
 548
 549        Returns:
 550            bool: True if the move was successful.
 551        """
 552        player = self.get_player(player_parity)
 553        self._apply_powerup_if_present(player_parity, self.cells[player.loc.r][player.loc.c])
 554
 555        if moves_this_turn >= 1:
 556            cost = player.stamina < GameConstants.EXTRA_MOVE_COST * moves_this_turn
 557            if player.stamina < cost:
 558                return False 
 559            player.stamina -= cost
 560
 561        if move.move_type == MoveType.ERASE:
 562            cost = GameConstants.ERASE_STEP_EXTRA_COST
 563            if player.stamina < cost:
 564                return False 
 565            player.stamina -= cost
 566            
 567        if move.move_type == MoveType.BEACON_TRAVEL:
 568            # use beacon to travel 
 569            target_loc = self._beacon_travel(player_parity, move, moves_this_turn)
 570            if target_loc is None:
 571                return False
 572        else:
 573            # don't use beacon to travel
 574            target_loc = player.loc + move.direction
 575            if self.oob(target_loc):
 576                return False
 577        
 578        # movement occurs here
 579        target_cell = self.cells[target_loc.r][target_loc.c]
 580        if target_cell.is_wall:
 581            return False
 582
 583        if self._resolve_collision(player_parity):
 584            return True
 585
 586        previous_loc = player.loc
 587        player.loc = target_loc
 588        
 589        # handle movement effects
 590        self._handle_erase_effects(player_parity, target_cell, move.move_type)
 591        self._apply_powerup_if_present(player_parity, target_cell)
 592        if move.place_beacon:
 593            self._place_beacon(player_parity, target_loc)
 594
 595        #  important that this comes before collision resolution
 596        if not player.clamp_stamina():
 597            return False
 598        
 599        
 600        
 601        return True

 603    def _beacon_travel(self, player_parity: int, move: Action.Move) -> bool:
 604        """
 605        Perform beacon-based teleportation.
 606
 607        Args:
 608            player_parity (int): Acting player's parity.
 609            move (Action.Move): Move containing beacon target.
 610
 611        Returns:
 612            Optional[Location]: Destination location, or None if invalid.
 613        """
 614        player = self.get_player(player_parity)
 615        if move.beacon_target is None:
 616            return None
 617        
 618        current_cell = self.cells[player.loc.r][player.loc.c]
 619        if current_cell.beacon_parity != player_parity: 
 620            return None
 621        
 622        dest = move.target
 623        if(self.oob(dest)):
 624            return None
 625        
 626        dest_cell = self.cells[dest.r][dest.c]
 627        if dest_cell.beacon_parity != player_parity:
 628            return None
 629        
 630        # consume the beacon at the origin TODO: do we want to allow choice of which beacon gets consumed? 
 631        current_cell.clear_beacon(GameConstants.BEACON_CONSUME_LEFTOVER_PAINT)
 632        player.beacon_count = max(0, player.beacon_count - 1)
 633
 634        return dest

 636    def _handle_erase_effects(self, player_parity: int, target_cell: CellState, step_type: MoveType) -> None:
 637        """
 638        Apply erase or weaken effects to a cell after movement.
 639
 640        Args:
 641            player_parity (int): Acting player's parity.
 642            target_cell (CellState): Cell affected by the step.
 643            step_type (MoveType): Type of movement step.

 646            return

 649        if step_type == MoveType.ERASE:
 650            target_cell.erase()
 651        else:
 652            target_cell.weaken_opponent(player_parity)
 653        now = target_cell.paint_value
 654
 655        if(now==0 and prev != 0):
 656            self._release_square(target_cell, Parity.parity_from_value(prev))

 563            cost = GameConstants.ERASE_STEP_EXTRA_COST

 584        if self._resolve_collision(player_parity):

 592        self._apply_powerup_if_present(player_parity, target_cell)
 593        if move.place_beacon:

 693        if cell.owner != player_parity:
 694            return False

 696            return False

 699            return False

 702        window_cells = origin.square_region(window_radius)
 703        
 704        friendly_cells: List[CellState] = []
 705        enemy_cells: List[CellState] = []
 706        opponent_parity = Parity.get_opponent_parity(player_parity)
 707
 708        for loc in window_cells:
 709            if self.oob(loc):
 710                continue
 711            candidate = self.cells[loc.r][loc.c]
 712            if(Parity.unowned(candidate.beacon_parity)):
 713                if Parity.owned(candidate.paint_value, player_parity):
 714                    friendly_cells.append(candidate)
 715                elif Parity.owned(candidate.paint_value, opponent_parity):
 716                    enemy_cells.append(candidate)
 717        
 718        if len(friendly_cells) < GameConstants.BEACON_REQUIREMENT_Q:
 719            return False
 720        
 721        player.stamina -= GameConstants.BEACON_COST
 722        
 723        for candidate in friendly_cells:
 724            candidate.paint_value = candidate.paint_value - player_parity
 725            if(candidate.paint_value == 0):
 726                self._release_square(candidate, player_parity)
 727
 728        # add to opponent's cells in the same area
 729        for candidate in enemy_cells:
 730            candidate.paint_value = max(
 731                -GameConstants.MAX_PAINT_VALUE,
 732                min(GameConstants.MAX_PAINT_VALUE, candidate.paint_value + opponent_parity)
